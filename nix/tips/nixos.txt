# cd to flake directory

# Update flake.lock

# Update:
sudo nix flake update --flake ~/nix/hosts/nixos/

# Apply the updates
sudo nixos-rebuild switch --cores 1 --flake ~/nix/hosts/nixos/ |& nom

# https://wiki.nixos.org/wiki/Cross_Compiling#Getting_Started:~:text=alternative
