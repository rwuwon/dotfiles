# nix on Debian:

# 1. Install and add user to group:
sudo apt install nix-setup-systemd
sudo adduser io nix-users

# 2. Enable channel:
nix-channel --add https://nixos.org/channels/nixpkgs-unstable nixpkgs
nix-channel --list
nix-channel --update

# 3. nix-shell will now work:
nix-shell -p vim

# 4. If using nix-env -iA:
mkdir -p ~/.config/nixpkgs
vim ~/.config/nixpkgs/config.nix
nix-env -iA nixpkgs.myPackages

export PATH=$PATH:~/.nix-profile/bin
